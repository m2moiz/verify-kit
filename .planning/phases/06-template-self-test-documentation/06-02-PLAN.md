    Canonical implementation:
    ```python
    from __future__ import annotations
    import secrets
    from typing import Annotated
    from fastapi import Depends, HTTPException, Request, status
    from fastapi.security import APIKeyHeader
    from harness.logging import log

    _api_key_header = APIKeyHeader(name="X-VerifyKit-Token", auto_error=False)
    _dev_fallback_warned = False

    def require_auth(
        request: Request,
        presented: Annotated[str | None, Depends(_api_key_header)],
    ) -> None:
        # In-dependency exclusion: /healthz must remain unauthenticated for docker
        # healthchecks and smoke tests. FastAPI app-level dependencies are NOT
        # overridable via route-level dependencies=[]; the short-circuit lives here.
        if request.url.path == "/healthz":
            return None

        settings = request.app.state.settings
        expected = settings.VERIFYKIT_AUTH_TOKEN

        if not expected:
            # Zero-config dev UX: log once, allow.
            if settings.ENV == "dev":
                global _dev_fallback_warned
                if not _dev_fallback_warned:
                    log.warning("auth.dev_fallback_no_token_configured")
                    _dev_fallback_warned = True
                return None
            raise HTTPException(
                status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
                detail="VERIFYKIT_AUTH_TOKEN not configured",
            )

        if presented is None or not secrets.compare_digest(
            presented.encode("utf-8"), expected.encode("utf-8")
        ):
            raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="invalid token")
        return None
    ```
